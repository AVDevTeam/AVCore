71 void PluginManager::unloadPlugin(std::string name)
72 {
73  // start of critical section
74  this->eventProcessingMutex.lock();
75  // this list will hold all callbacks that were registered
76  // by the unloading plugin.
77  std::list<std::pair<priorityMap*, int>> toDelete;
78 
79  IPlugin* plugin = this->loadedPlugins[name];
80  // iterate over callbackMap and gather all callbacks of the unloadin plugin
81  for (eventsMap::iterator it = this->callbacksMap.begin(); it != this->callbacksMap.end(); it++)
82  for (priorityMap::iterator it2 = (*it).second->begin(); it2 != (*it).second->end(); it2++)
83  {
84  callback curCallback = (*it2).second;
85  if (curCallback.second == plugin)
86  toDelete.push_back(std::pair<priorityMap*, int>((*it).second, (*it2).first));
87  }
88  // do the actual erasing of the callbacks from the map.
89  for (std::list<std::pair<priorityMap*, int>>::iterator it = toDelete.begin(); it != toDelete.end(); it++)
90  (*it).first->erase((*it).second);
91  this->loadedPlugins.erase(name);
92  // free plugin DLL
93  HMODULE pluginModule = plugin->getModule();
94  // here the plugin should free the memory (including
95  // IPlugin instance that was returned from DLL entry point).
96  plugin->deinit();
97  FreeLibrary(pluginModule);
98  // leaving critical section
99  this->eventProcessingMutex.unlock();
100 }

125 Method description:
126  This method is used in event-streams listeners to handle events.
127  First the eventParsers are used to represent event as an
128  instance of AvEvent class.
129  Method iterates over callback map for the given event type
130  from the callback with highest priority (the smallest int).

132  If callback returns BlockStatus than the event processing stops.
133  If not, the next callback for this event is called.
